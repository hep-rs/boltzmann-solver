pub trait PowAssign<Rhs> {
    fn pow_assign(&mut self, rhs: Rhs);
}

Compound power operation and assignment.

Examples

use rug::ops::PowAssign;
struct U(u32);
impl PowAssign<u16> for U {
    fn pow_assign(&mut self, rhs: u16) {
        self.0 = self.0.pow(rhs as u32);
    }
}
let mut u = U(5);
u.pow_assign(2_u16);
assert_eq!(u.0, 25);
